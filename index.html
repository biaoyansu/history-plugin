Biao-History

安装

依赖安装

<link rel="stylesheet" href="history.css"> 
<!--引入history.css-->

<script src="history.min.js"></script> 
<!--引入history.js-->

设置根元素

<div id="history-container"></div>

基础用法

bh.init({
  el: '#history-container',
});

bh.add('Yo');
bh.add('Yoo');

配置

bh.init({
  el: '#selector', 
// 选择器

  on_click: function() {} 
// 点击回调

  on_delete: function() {} 
// 删除回调

});

bh.add('Yo');
bh.add('Yoo');

API

数据操作

bh.add('Yo') 
// 添加一条历史记录

bh.remove('Yo') 
// 搜索并删除一条历史记录

bh.clear() 
// 清空所有历史记录

视图操作

bh.show() 
// 显示

bh.hide() 
// 隐藏

bh.is_visible() 
// 是否可见
